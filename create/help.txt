create [language] [project name] ...flags

Languages:
	py: Python
	c++ / cpp: C++
	node: Node.js
	go: Golang
	arduino: Arduino
	flutter: Flutter
	workspace: VScode workspace
	help: Display current help message

Flags:
	-g: Create a remote git repository
	-l: Make the repository local, not remote
	-v: Make the remote repository publicly visible
	-r: Create a README.md file
